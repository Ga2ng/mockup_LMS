    <script>
    (function(){
      const box=document.getElementById('scoreBox'); if(!box) return;
      const cls=document.getElementById('v50ScoreClassFilter');
      const q=document.getElementById('v50ScoreSearch');
      const count=document.getElementById('v50ScoreCount');
      const rows=[...box.querySelectorAll('tr.v50-score-row')];
      function norm(s){return String(s||'').toLowerCase().trim();}
      function apply(){
        const c=norm(cls&&cls.value), term=norm(q&&q.value); let shown=0;
        rows.forEach(r=>{
          const okClass=!c || norm(r.dataset.kelas)===c;
          const okText=!term || norm(r.dataset.search).indexOf(term)>=0;
          const ok=okClass && okText;
          r.style.display=ok?'':'none'; if(ok) shown++;
        });
        if(count) count.textContent=shown;
      }
      cls&&cls.addEventListener('change',apply);
      q&&q.addEventListener('input',apply);
      apply();
    })();
    </script>
